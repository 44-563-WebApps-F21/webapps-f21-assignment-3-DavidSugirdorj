Algorithm

Given a source vertex s from a set of vertices V in a weighted graph where all its edge weights

w(u, v) are non-negative, find the shortest path weights d(s, v)

from source s for all vertices v present in the graph.

const int INF = 1000000000;    
        vector>> adj;
        
        void dijkstra(int s, vector & d, vector & p) {
            int n = adj.size();
            d.assign(n, INF);
            p.assign(n, -1);
            vector u(n, false);
        
            d[s] = 0;
            for (int i = 0; i < n; i++) {
                int v = -1;
                for (int j = 0; j < n; j++) {
                    if (!u[j] && (v == -1 || d[j] < d[v]))
                        v = j;
                }
        
                if (d[v] == INF)
                    break;
        
                u[v] = true;
                for (auto edge : adj[v]) {
                    int to = edge.first;
                    int len = edge.second;
        
                    if (d[v] + len < d[to]) {
                        d[to] = d[v] + len;
                        p[to] = v;
                    }
                }
            }
        }
